这是百度的响应格式 ↓
let input = document.querySelector('input');    //这里获取的是搜索栏的输入框

// 从 Baidu 获取热搜关键词
function inp(keyWord) {
    let s = document.createElement('script');
    s.src = `https://www.baidu.com/sugrec?pre=1&p=3&ie=utf-8&json=1&prod=pc&from=pc_web&sugsid=1454,21100,29522,29720,29568,29221,26350&wd=${keyWord}&req=2&csor=3&pwd=wi&cb=search&_=1568883721394`;  
    // 这段超长的 url 取自百度
    document.body.appendChild(s);
}

// 处理获取到的数据
function search(data) {
    let ul = document.querySelector('ul');

    // 对数据的处理，详见上图的响应格式
    if (data.g === undefined) return;
    let liArr = data.g.map(word => `<li class="hot">${word.q}</li>`);
    ul.innerHTML = liArr.join('') + `<li><a class="feedBack">反馈</a></li>`;
}

input.oninput = function () {
    document.querySelector('ul').style.display = "block";
    if (input.value == '') document.querySelector('ul').innerHTML = '';
    inp(input.value);   // 调用 inp() 方法 , 利用 JSONP 获取热搜关键词
}
input.onblur = function () {
    // 搜索栏失去焦点时隐藏热搜关键词列表
    setTimeout(() => {
        document.querySelector('ul').style.display = "none";
    }, 100);
}

// logo
document.querySelector(".logo img").onclick = function () {
    location.href = "#";
}

// 以下为搜索功能
document.querySelector("input[type='submit']").onclick = function () {
    let word = input.value;
    location.href = "https://www.baidu.com/s?wd=" + word;
}
document.querySelector("ul").onclick = function (evt) {
    console.log(evt.target.className);

    if (evt.target.className === 'hot') {
        let word = evt.target.innerHTML;
        location.href = "https://www.baidu.com/s?wd=" + word;
    }
}
